CREATE PROCEDURE spUpdateLoans
@loanId NCHAR,
@copyId NCHAR,
@BID NCHAR,
@LDate DATE,
@RDate DATE
AS
BEGIN
	SET NOCOUNT ON;
	Update dbo.Loan
	SET 
		[CopyID]=@copyId,
		[BID]=@BID,
		[LDate]=@LDate,
		[RDate]=@RDate
	WHERE [LoanID]=@loanId
END
GO
